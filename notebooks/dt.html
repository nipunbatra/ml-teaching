import pandas as pd
import numpy as np
def entropy(s):
    """
    s: pd.Series of dtype category
    returns: entropy
    """
    p = s.value_counts()/len(s)
    entropy_df = pd.DataFrame({"p":p, "-plogp":-p*np.log2(p)})
    return entropy_df['-plogp'].sum()

d['feature_names']

 'petal width (cm)']

colours = ['blue', 'red', 'green']
species = ['I. setosa', 'I. versicolor', 'I. virginica']
for i in range(0, 3):

    plt.scatter(        
        df_[:, 0],        
        df_[:, 1],
        color=colours[i],        
        alpha=0.5,        
        label=species[i] ,
        s=10
    )
